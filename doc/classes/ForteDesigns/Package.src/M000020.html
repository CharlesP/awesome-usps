    def grams(options={})
      case options[:type]
      when *[nil,:actual]: @grams ||= ounces(options) * GRAMS_IN_AN_OUNCE
      when *[:volumetric,:dimensional]: @volumetric_grams ||= centimetres(:box_volume) / 6.0
      when :billable: [grams,grams(:type => :volumetric)].max
      end
    end
